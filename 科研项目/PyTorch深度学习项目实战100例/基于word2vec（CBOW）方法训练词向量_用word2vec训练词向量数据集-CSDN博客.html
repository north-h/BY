一、基于word2vec（CBOW）方法训练词向量

word2vec是一种将word转为向量的方法，其包含两种算法，分别是skip-gram和CBOW，它们的最大区别是skip-gram是通过中心词去预测中心词周围的词，而CBOW是通过周围的词去预测中心词。

本项目使用CBOW来训练一个word2vec模型，得到语料库中的词向量表示。

二、数据集介绍

本项目使用的预料数据是使用的英文小说呼啸山庄，由于版权问题这里就不提供数据集了，可以自己去百度一些英文文章或者爬一些小说作为预料数据进行训练。

三、CBOW介绍

word2vec是一种将word转为向量的方法，其包含两种算法，分别是skip-gram和CBOW，它们的最大区别是skip-gram是通过中心词去预测中心词周围的词，而CBOW是通过周围的词去预测中心词。

这个word2vec的方法是在2013年的论文《Efficient Estimation of Word Representations inVector Space》中提出的，作者来自google

文章下载链接：https://arxiv.org/pdf/1301.3781.pdf

  • 输入层：一个形状为C×V的one-hot张量，其中代表上线文中词的个数，通常是一个偶数，表示词表大小，该张量的每一行都是一个上下文词的one-hot向量表示，比如“Pineapples,are, and, yellow”。

  • 隐藏层：输入张量和word embedding W进行矩阵乘法，就会得到一个形状为×N的张量。综合考虑上下文中所有词的信息去预测中心词，因此将上下文中个词相加得一个1×的向量，是整个上下文的一个隐含表示。

  • 输出层：将隐藏层得到的1×的向量乘以该×的参数张量，得到了一个形状为1×的向量。最终，1×的向量代表了使用上下文去预测中心词，再经过softmax函数的归一化，即得到了对中心词的预测概率。

四、定义CBOW网络模型

# 定义CBOW模型
class CBOW(nn.Module):
    def __init__(self, vocab_size, embedding_dim):
        super(CBOW, self).__init__()
        # 嵌入层
        self.embeddings = nn.Embedding(vocab_size, embedding_dim)
        # 映射层
        self.project = nn.Linear(embedding_dim, 128)
        # 输出层
        self.fc = nn.Linear(128, vocab_size)

    def forward(self, x):
#         print(x.shape) # torch.Size([32, 4])
        embeds = self.embeddings(x) # torch.Size([32, 4, 256])

        embeds_sum = embeds.sum(dim=1) # torch.Size([32, 256])
        
        x_project = self.project(embeds_sum)
        x_project = F.relu(x_project)

        output = self.fc(x_project)
        nll_prob = F.log_softmax(output, dim=-1)
        return nll_prob

六、训练模型

# 5.训练模型
model = CBOW(vocab_size, embedding_dim).to(device)
optimizer = optim.Adam(model.parameters(), lr=lr) # 优化器
loss_function = nn.NLLLoss() # 损失函数

# 存储损失的集合
losses = []

for epoch in trange(epochs):
    total_loss = 0
    train_bar = tqdm(train_loader)
    for context, target in train_bar:
        model.zero_grad()
        train_predict = model(context)
        
        loss = loss_function(train_predict, target.long())

        loss.backward()

        optimizer.step()
        total_loss += loss.item()
        
    losses.append(total_loss)
    
    # 打印信息
    print("【EPOCH: 】%s" % str(epoch + 1))
    print("训练损失为%s" % (str(total_loss)))
    
print("losses-=", losses)

import torch
import numpy as np
import torch.nn as nn
import torch.nn.functional as F
import torch.optim as optim
from matplotlib import pyplot as plt
from sklearn.decomposition import PCA
from tqdm import tqdm, trange

# 1.读取预料数据
with open('./english.txt', 'r', encoding='utf-8') as f:
    data = f.read()[: 1000]
raw_text = data.split()
 
print("raw_text=", raw_text)

learning_rate = 0.003 # 学习率
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu') # 设备
context_size = 2 # 窗口大小，利用前后各2各词预测当前词
embedding_dim = 256 # 词嵌入维度
epochs = 10 # 训练轮数
batch_size = 32 # 批次大小

# 将对应的词转成索引
def make_context_vector(contexts, labels, word_to_ix):
    x = [[word_to_idx[w] for w in context] for context in contexts]
    y = [word_to_idx[w] for w in labels]
    x = np.array(x)
    y = np.array(y)
    return x, y

# 2.把所有词集合转成dict
vocab = set(raw_text)
vocab_size = len(vocab)

word_to_idx = {word: i for i, word in enumerate(vocab)}
idx_to_word = {i: word for i, word in enumerate(vocab)}

# 3.形成训练数据，即{[w1,w2,w4,w5], "w3"}这样形式
data = []
label = []

for i in range(context_size, len(raw_text) - context_size):
    context = [raw_text[i - 2], raw_text[i - 1],
               raw_text[i + 1], raw_text[i + 2]]
    target = raw_text[i]
    data.append(context)
    label.append(target)

print(data[:5])

data, label = make_context_vector(data, label, word_to_idx)

# 4.形成迭代器
dataset = torch.utils.data.TensorDataset(torch.tensor(data), torch.tensor(label))
train_loader = torch.utils.data.DataLoader(dataset=dataset, batch_size=batch_size)

# 定义CBOW模型
class CBOW(nn.Module):
    def __init__(self, vocab_size, embedding_dim):
        super(CBOW, self).__init__()
        # 嵌入层
        self.embeddings = nn.Embedding(vocab_size, embedding_dim)
        # 映射层
        self.project = nn.Linear(embedding_dim, 128)
        # 输出层
        self.fc = nn.Linear(128, vocab_size)

    def forward(self, x):
#         print(x.shape) # torch.Size([32, 4])
        embeds = self.embeddings(x) # torch.Size([32, 4, 256])

        embeds_sum = embeds.sum(dim=1) # torch.Size([32, 256])
        
        x_project = self.project(embeds_sum)
        x_project = F.relu(x_project)

        output = self.fc(x_project)
        nll_prob = F.log_softmax(output, dim=-1)
        return nll_prob

# 5.训练模型
model = CBOW(vocab_size, embedding_dim).to(device)
optimizer = optim.Adam(model.parameters(), lr=lr) # 优化器
loss_function = nn.NLLLoss() # 损失函数

# 存储损失的集合
losses = []

for epoch in trange(epochs):
    total_loss = 0
    train_bar = tqdm(train_loader)
    for context, target in train_bar:
        model.zero_grad()
        train_predict = model(context)
        
        loss = loss_function(train_predict, target.long())

        loss.backward()

        optimizer.step()
        total_loss += loss.item()
        
    losses.append(total_loss)
    
    # 打印信息
    print("【EPOCH: 】%s" % str(epoch + 1))
    print("训练损失为%s" % (str(total_loss)))
    
print("losses-=", losses)

# 6.测试结果
# 用 ['I', 'have', 'returned', 'from'] 预测中间词 'just'
context = ['I', 'have', 'returned', 'from']

# 将其转成tensor
context = torch.tensor([word_to_idx[w] for w in context])

# 7.预测结果
